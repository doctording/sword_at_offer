# DFS求解

注意python变量域，这个问题要整理

```python
# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    # 返回二维列表，内部每个列表表示找到的路径
    def FindPath(self, root, expectNumber):
        ans = []

        if root == None :
            return ans

        rs = []
        rs.append(root.val)

        # 内部方法
        def dfs(root, sums, rs) :

            if root.left == None and root.right == None :
                if sums == 0 :
                    #print rs
                    tmp = []
                    tmp += rs
                    ans.append(tmp)
                return None

            if root.left != None :
                if root.left.val <= sums :
                    rs.append(root.left.val)
                    dfs(root.left, sums - root.left.val, rs)
                    rs.pop(-1) # 回溯

            if root.right != None :
                if root.right.val <= sums :
                    rs.append(root.right.val)
                    dfs(root.right, sums - root.right.val, rs)
                    rs.pop(-1) # 回溯

        dfs(root, expectNumber - root.val, rs)

        return ans
```

# BFS求解

```python
# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    # 返回二维列表，内部每个列表表示找到的路径
    def FindPath(self, root, expectNumber):
        # write code here
        ans = []
        if root == None :
            return ans

        que = []
        que_size = 1
        que.append(root)

        queS = []
        queS_size = 1
        queS.append(root.val)

        mp = {}
        mp[root] = None

        while que_size != 0 :
            node = que[0]
            que.pop(0)
            que_size -= 1

            nodeSum = queS[0]
            queS.pop(0)
            queS_size -= 1

            if node.left == None and node.right == None and nodeSum == expectNumber :
                li = []
                tmp = node
                while tmp != None :
                    li.insert(0, tmp.val)
                    tmp = mp[tmp]
                ans.insert(0, li) # 牛客网这里对结果顺序有要求？

            if node.left != None :
                nextNode = node.left
                mp[nextNode] = node
                que.append(nextNode)
                queS.append(nodeSum + nextNode.val)
                que_size += 1
                queS_size += 1

            if node.right != None :
                nextNode = node.right
                mp[nextNode] = node
                que.append(nextNode)
                queS.append(nodeSum + nextNode.val)
                que_size += 1
                queS_size += 1

        return ans
```
