```python
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def FindKthToTail(self, head, k):
        # write code here
        # 单链表为空 或者 k 大于 单链表的长度都返回 None
        if head == None :
            return None

        i = 0
        head2 = head
        while i < k :
            if head2 == None :
                return None
            head2 = head2.next
            i += 1

        rs = head
        while head2 != None :
            rs = rs.next
            head2 = head2.next

        return rs
```
