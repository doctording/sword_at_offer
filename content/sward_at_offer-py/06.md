# 解法1

* ```O(n)```

```python
# -*- coding:utf-8 -*-
class Solution:
    def minNumberInRotateArray(self, rotateArray):
        # write code here
        n = len(rotateArray)
        if n == 0 :
            return 0

        i = 0
        j = n - 1
        while i < j :
            if rotateArray[i] >= rotateArray[j] :
                i = i + 1
            else :
                j = j - 1
        return rotateArray[i]
```

# 二分法（有特殊情况）

```python
# -*- coding:utf-8 -*-
class Solution:

    def minNumberInRotateArray(self, rotateArray):
        # write code here
        n = len(rotateArray)
        if n == 0 :
            return 0

        i = 0
        j = n - 1
        while i < j :
            mid = (j - i) / 2 + i
            if rotateArray[mid] > rotateArray[i] :
                i = mid
            elif rotateArray[mid] < rotateArray[i] :
                j = mid
            else:
                if i == j : 
                    return rotateArray[i]
                return min(rotateArray[i:j+1]) # list i:j的含义？
        return rotateArray[i]
```

            return 0

        j = n - 1

        while i <= j and rotateArray[i] >= rotateArray[j] :
            if j - i == 1 :
                return rotateArray[j]
            mid = (j - i) / 2 + i
            if rotateArray[i] <= rotateArray[mid] :
                i = mid
            if rotateArray[mid] <= rotateArray[j] :
                j = mid
        return min(rotateArray[i: j + 1])
