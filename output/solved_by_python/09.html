数学公式推导求解

# -*- coding:utf-8 -*-
class Solution:
    def jumpFloorII(self, number):
        # write code here
        # fn = fn-1 + fn-2 + ... + f1 + 1
        # fn = 2^(n - 2) * (f1 + 1)
        if number == 1 :
            return 1
        if number == 2 :
            return 2

        n = 1
        n = n << (number - 2)
        return n * (1 + 1)
