            return 0

            return 1

        i = 3
        while i <= n :
            tmp = t1 + t2
            t1 = t2
            t2 = tmp

        return t2

# -*- coding: UTF-8 -*-

def multi(m1, m2) :
    m = [[0,0],[0,0]]
    for i in range(0,2) :
        for j in range(0,2) :
            for k in range(0, 2) :
                m[i][j] += m1[i][k] * m2[k][j]
    return m

def multiN(m, n) :
    rs = [[1, 0],[0, 1]]
    while n > 0 :
        if n & 1 == 1 :
            rs = multi(rs, m)
        m = multi(m, m)
        n = n >> 1
    return rs

class Solution:
    def Fibonacci(self, n):
        # write code here
        if n == 0 :
            return 0
        if n == 1 or n == 2 :
            return 1
        m = [[1,1],[1,0]]
        # [fn fn-1] = [f1 f0] * matrix^(n-1)
        rs = multiN(m, n-2)
        ans = 0
        for i in range(0, 2) :
            ans += rs[i][0]
        return ans
