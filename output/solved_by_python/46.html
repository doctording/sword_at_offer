class Solution:
    def LastRemaining_Solution(self, n, m):
        if n == 0 or m == 0 :
            return -1

        li = range(0, n)

        li_size = n
        cur_num = 0
        cur_index = 0
        while li_size > 1 :
            cur_index = (cur_index + m - 1) % li_size    # 下一个位置
            print li[cur_index]
            li.pop(cur_index)

            if cur_index == li_size - 1:
                cur_index = 0
            li_size -= 1

        return li[0]

            while i_cnt > 0:

                    if cur_next == n - 1:
                        cur_next = 0
                    else:
                        cur_next += 1
                else:
                    i_cnt -= 1
                    if cur_next == n - 1:
                        cur_next = 0
                    else:
                        cur_next += 1
                    while mp.has_key(cur_next):
                        if cur_next == n - 1:
                            cur_next = 0
                        else:
                            cur_next += 1

                mp[cur_next] = 1
                cnt -= 1

                if cur_next == n - 1:
                    cur_next = 0
                else:
                    cur_next += 1

                    if cur_next == n - 1:
                        cur_next = 0
                    else:
                        cur_next += 1

                cur = cur_next
